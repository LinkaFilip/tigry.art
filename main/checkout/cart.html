import { products } from '../data/products.js'; const cart = JSON.parse(localStorage.getItem('cart')) || []; const cartContainer = document.getElementById('cart-items'); cart.forEach(item => { const product = products.find(p => p.id === item.id); const html = `

${product.name}

${product.description}

${item.quantity} × ${product.price / 100} €

${(product.price * item.quantity) / 100} €

`; cartContainer.innerHTML += html; });
